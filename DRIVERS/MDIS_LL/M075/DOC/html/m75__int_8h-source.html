00217 /* Macros for accessing SCC registers */
00218 #define WRITE_SCC_COMM( ma,ch,val )                     \
00219     do {                                                \
00220         if( ch == 0 ){  /* write to channel A */        \
00221             MWRITE_D8( ma, COMM_REG_A, val );           \
00222             IDBGWRT_6( (DBH, "Write COMM_REG_A %02X\n", val) ); \
00223         } else {        /* write to channel B */        \
00224             MWRITE_D8( ma, COMM_REG_B, val );           \
00225             IDBGWRT_6( (DBH, "Write COMM_REG_B %02X\n", val) ); \
00226         }                                               \
00227     } while( 0 )
00228 
00229 #define READ_SCC_COMM( ma,ch,val )                      \
00230     do {                                                \
00231         if( ch == 0 ){  /* write to channel A */        \
00232             val = MREAD_D8( ma, COMM_REG_A );   \
00233             IDBGWRT_6( (DBH, "Read  COMM_REG_A %02X\n", val) ); \
00234         } else {        /* write to channel B */        \
00235             val = MREAD_D8( ma, COMM_REG_B );   \
00236             IDBGWRT_6( (DBH, "Read  COMM_REG_B %02X\n", val) ); \
00237         }                                               \
00238     } while( 0 )
00239 
00240 #define WRITE_SCC_REG( ma,ch,reg,val )                              \
00241     do {                                                            \
00242         OSS_IRQ_STATE irqStateWAcc;                                 \
00243         irqStateWAcc = OSS_IrqMaskR(llHdl->osHdl, llHdl->irqHdl);   \
00244         WRITE_SCC_COMM( ma,ch,reg );                                \
00245         WRITE_SCC_COMM( ma,ch,val );                                \
00246         if( reg == M75_R08) {   /* data register, do nothing more */\
00247         } else if( reg != M75_R07 ) { /* normal SCC register */     \
00248             /* mirror it */                                         \
00249             *llHdl->chan[ch].sccRegs.wrp[reg] = val;                \
00250         } else if( (reg == M75_R07) &&                              \
00251                    (llHdl->chan[ch].sccRegs.wr15 & M75_SCC_WR15_WR7P_EN) ) {/* WR7P */  \
00252             llHdl->chan[ch].sccRegs.wr7p = val;                     \
00253         } else if( reg == M75_R07 ){                     /* WR07 */ \
00254             llHdl->chan[ch].sccRegs.wr07 = val;                     \
00255         }                                                           \
00256         OSS_IrqRestore( llHdl->osHdl, llHdl->irqHdl, irqStateWAcc ); \
00257         IDBGWRT_5( (DBH, "Write ch %d SCC Reg %02d: %02X\n", ch, reg, val) ); \
00258     } while( 0 )
